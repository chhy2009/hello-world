531inline void TFProfObserver::dump(std::ostream& os) const {

533 size_t first;

535 for(first = 0; first<_timeline.segments.size(); ++first) {

541 // not timeline data to dump
542 if(first == _timeline.segments.size()) {

559 os << ',';
